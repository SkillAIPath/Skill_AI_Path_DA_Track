-- Step 3A: Comprehensive customer profiling with loan performance -- BUSINESS OBJECTIVE: Create complete customer profiles for risk analysis -- CRISIS CONTEXT: Need to identify which customer types are driving portfolio losses SELECT TOP 15 c.customer_id, c.full_name, dc.city_name, ds.state_name, dc.tier_classification, -- Customer demographics and financials c.annual_income, c.cibil_score, c.employment_type, c.age, c.education_level, -- Loan portfolio analysis COUNT(l.loan_id) AS total_loans, SUM(l.loan_amount) AS total_exposure, ROUND(AVG(l.loan_amount), 0) AS avg_loan_size, -- Loan status breakdown COUNT(CASE WHEN l.loan_status = 'Active' THEN 1 END) AS active_loans, COUNT(CASE WHEN l.loan_status = 'Defaulted' THEN 1 END) AS defaulted_loans, COUNT(CASE WHEN l.loan_status = 'Overdue' THEN 1 END) AS overdue_loans, -- Performance indicators ROUND( COUNT(CASE WHEN l.loan_status = 'Defaulted' THEN 1 END) * 100.0 / NULLIF(COUNT(l.loan_id), 0), 2 ) AS personal_default_rate FROM customers c INNER JOIN dim_city dc ON c.city_id = dc.city_id INNER JOIN dim_state ds ON dc.state_id = ds.state_id INNER JOIN loans l ON c.customer_id = l.customer_id WHERE l.disbursement_date IS NOT NULL GROUP BY c.customer_id, c.full_name, dc.city_name, ds.state_name, dc.tier_classification, c.annual_income, c.cibil_score, c.employment_type, c.age, c.education_level HAVING COUNT(l.loan_id) >= 2 -- Risk customers with multiple loans ORDER BY defaulted_loans DESC, total_exposure DESC;

Analytical Purpose: Segment customers by income brackets to identify which income groups pose highest default risk, challenging traditional underwriting assumptions.

-- Step 3B: Income-based risk analysis with sophisticated segmentation -- BUSINESS OBJECTIVE: Challenge assumption that higher income = lower risk -- CRISIS DISCOVERY: High-income segments may have highest default rates SELECT -- Income segmentation logic CASE WHEN c.annual_income >= 1200000 THEN 'Ultra High (‚Çπ12L+)' WHEN c.annual_income >= 800000 THEN 'High Income (‚Çπ8-12L)' WHEN c.annual_income >= 500000 THEN 'Middle Income (‚Çπ5-8L)' WHEN c.annual_income >= 300000 THEN 'Lower Middle (‚Çπ3-5L)' ELSE 'Low Income (<‚Çπ3L)' END AS income_segment, -- Portfolio metrics by segment COUNT(DISTINCT c.customer_id) AS customers_in_segment, COUNT(l.loan_id) AS total_loans, CONCAT('‚Çπ', ROUND(SUM(l.loan_amount) / 10000000, 2), ' Cr') AS segment_portfolio, ROUND(AVG(l.loan_amount), 0) AS avg_loan_amount, -- Risk analysis COUNT(CASE WHEN l.loan_status = 'Defaulted' THEN 1 END) AS defaulted_loans, ROUND( COUNT(CASE WHEN l.loan_status = 'Defaulted' THEN 1 END) * 100.0 / NULLIF(COUNT(l.loan_id), 0), 2 ) AS segment_default_rate, -- Financial impact CONCAT('‚Çπ', ROUND( SUM(CASE WHEN l.loan_status = 'Defaulted' THEN l.loan_amount ELSE 0 END) / 10000000, 2 ), ' Cr' ) AS segment_losses, -- Business risk classification CASE WHEN COUNT(CASE WHEN l.loan_status = 'Defaulted' THEN 1 END) * 100.0 / NULLIF(COUNT(l.loan_id), 0) >= 20 THEN 'üî¥ HIGH RISK SEGMENT' WHEN COUNT(CASE WHEN l.loan_status = 'Defaulted' THEN 1 END) * 100.0 / NULLIF(COUNT(l.loan_id), 0) >= 12 THEN 'üü° MODERATE RISK' ELSE 'üü¢ ACCEPTABLE RISK' END AS segment_risk_level FROM customers c STEP 3C: CIBIL Score Risk Analysis

Analytical Purpose: Analyze default patterns across CIBIL score ranges to identify if credit scores are accurate predictors or if underwriting models need complete overhaul.

-- Step 3C: CIBIL score effectiveness analysis -- BUSINESS OBJECTIVE: Test if CIBIL scores actually predict default risk -- CRISIS HYPOTHESIS: Good CIBIL customers may be defaulting unexpectedly SELECT -- CIBIL score segmentation CASE WHEN c.cibil_score >= 800 THEN 'Excellent (800+)' WHEN c.cibil_score >= 750 THEN 'Very Good (750-799)' WHEN c.cibil_score >= 700 THEN 'Good (700-749)' WHEN c.cibil_score >= 650 THEN 'Fair (650-699)' WHEN c.cibil_score >= 600 THEN 'Poor (600-649)' ELSE 'Very Poor (<600)' END AS cibil_segment, -- Portfolio and customer metrics COUNT(DISTINCT c.customer_id) AS customers, COUNT(l.loan_id) AS total_loans, ROUND(AVG(c.annual_income), 0) AS avg_income, CONCAT('‚Çπ', ROUND(SUM(l.loan_amount) / 10000000, 2), ' Cr') AS segment_portfolio, -- Risk performance analysis COUNT(CASE WHEN l.loan_status = 'Defaulted' THEN 1 END) AS defaults, ROUND( COUNT(CASE WHEN l.loan_status = 'Defaulted' THEN 1 END) * 100.0 / NULLIF(COUNT(l.loan_id), 0), 2 ) AS default_rate, -- Financial impact by CIBIL segment CONCAT('‚Çπ', ROUND( SUM(CASE WHEN l.loan_status = 'Defaulted' THEN l.loan_amount ELSE 0 END) / 10000000, 2 ), ' Cr' ) AS losses, -- CIBIL effectiveness assessment CASE WHEN c.cibil_score >= 700 AND COUNT(CASE WHEN l.loan_status = 'Defaulted' THEN 1 END) * 100.0 / NULLIF(COUNT(l.loan_id), 0) > 15 THEN 'üö® CIBIL MODEL FAILURE' WHEN COUNT(CASE WHEN l.loan_status = 'Defaulted' THEN 1 END) * 100.0 / NULLIF(COUNT(l.loan_id), 0) > 20 THEN 'üî¥ CRISIS SEGMENT' WHEN COUNT(CASE WHEN l.loan_status = 'Defaulted' THEN 1 END) * 100.0 / NULLIF(COUNT(l.loan_id), 0) > 12 THEN 'üü° HIGH RISK' ELSE 'üü¢ ACCEPTABLE' END AS model_effectiveness FROM customers c INNER JOIN loans l ON c.customer_id = l.customer_id WHERE l.disbursement_date IS NOT NULL AND c.cibil_score IS NOT NULL GROUP BY CASE WHEN c.cibil_score >= 800 THEN 'Excellent (800+)' WHEN c.cibil_score >= 750 THEN 'Very Good (750-799)' WHEN c.cibil_score >= 700 THEN 'Good (700-749)' WHEN c.cibil_score >= 650 THEN 'Fair (650-699)' WHEN c.cibil_score >= 600 THEN 'Poor (600-649)' ELSE 'Very Poor (<600)' END ORDER BY AVG(c.cibil_score) DESC;

-- STEP 3: Complete Customer Risk Segmentation Dashboard -- BUSINESS OBJECTIVE: Executive customer risk intelligence for targeted interventions -- CRISIS CONTEXT: Identify high-value customers requiring immediate attention WITH customer_risk_profile AS ( SELECT c.customer_id, c.full_name, dc.city_name, ds.state_name, -- Customer characteristics c.annual_income, c.cibil_score, c.employment_type, c.age, -- Loan portfolio metrics COUNT(l.loan_id) AS total_loans, SUM(l.loan_amount) AS total_exposure, COUNT(CASE WHEN l.loan_status = 'Defaulted' THEN 1 END) AS defaulted_loans, c.customer_id = l.customer_id WHERE l.disbursement_date IS NOT NULL GROUP BY c.customer_id, c.full_name, dc.city_name, ds.state_name, c.annual_income, c.cibil_score, c.employment_type, c.age HAVING COUNT(l.loan_id) >= 2 -- Focus on multi-loan customers ) SELECT TOP 20 crp.customer_id, crp.full_name, crp.city_name, crp.state_name, -- Customer profiling CONCAT('‚Çπ', ROUND(crp.annual_income /
